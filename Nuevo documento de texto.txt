package com.empresa.proyecto.config;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.web.SecurityFilterChain;
import org.springframework.security.oauth2.server.resource.authentication.JwtAuthenticationConverter;
import org.springframework.security.oauth2.server.resource.authentication.JwtGrantedAuthoritiesConverter;

@Configuration
public class SecurityConfig {

    @Value("${app.env:dev}")
    private String appEnv;  // lee la propiedad de application.properties

    @Bean
    public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {

        http.csrf(csrf -> csrf.disable());

        http.authorizeHttpRequests(auth -> auth
            // Endpoints públicos
            .requestMatchers("/api/public/**").permitAll()
            
            // Swagger
            .requestMatchers("/swagger-ui.html", "/swagger-ui/**", "/v3/api-docs/**")
                .access(access -> {
                    if ("dev".equals(appEnv) || "local".equals(appEnv)) {
                        return access.permitAll();      // Swagger libre
                    } else {
                        return access.authenticated();  // Swagger protegido en prod
                    }
                })

            // Todo lo demás requiere JWT
            .anyRequest().authenticated()
        );

        // Configuración de JWT para Azure AD
        http.oauth2ResourceServer(oauth2 -> oauth2
            .jwt(jwt -> jwt.jwtAuthenticationConverter(jwtAuthenticationConverter()))
        );

        return http.build();
    }

    // JwtAuthenticationConverter para mapear roles o scopes de Azure AD
    @Bean
    public JwtAuthenticationConverter jwtAuthenticationConverter() {
        JwtGrantedAuthoritiesConverter authoritiesConverter = new JwtGrantedAuthoritiesConverter();
        authoritiesConverter.setAuthorityPrefix("ROLE_");      // prefijo para roles
        authoritiesConverter.setAuthoritiesClaimName("roles"); // claim del token (roles o scp)

        JwtAuthenticationConverter converter = new JwtAuthenticationConverter();
        converter.setJwtGrantedAuthoritiesConverter(authoritiesConverter);
        return converter;
    }
}
